SELECT  T1.ID       AS 'ID'
    ,   (SELECT FISH_NAME FROM FISH_NAME_INFO WHERE FISH_TYPE = T1.FISH_TYPE) AS 'FISH_NAME'
    ,   T1.LENGTH   AS 'LENGTH'
FROM    FISH_INFO T1
JOIN (
    SELECT  FISH_TYPE
        ,   MAX(TT1.LENGTH) AS 'MAX_LENGTH'
    FROM    FISH_INFO TT1
    GROUP BY
            FISH_TYPE
) T2
ON      T2.FISH_TYPE    = T1.FISH_TYPE
    AND T2.MAX_LENGTH   = T1.LENGTH
-- WHERE   T1.LENGTH   = T2.MAX_LENGTH
ORDER BY
        T1.ID   ASC
